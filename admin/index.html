Conectar Analytics

Cole a URL do Worker para exibir métricas de visitas do site. As credenciais ficam seguras no Cloudflare.

URL do Worker criado no Cloudflare Workers & Pages

▸ Código atualizado do Worker (com credenciais seguras)
  1. No Worker, vá em Edit code, apague tudo e cole o código abaixo
  2. Clique em Deploy
  3. Vá em Settings → Variables and Secrets e adicione CF_ZONE_ID e CF_API_TOKEN
export default {
  async fetch(request, env) {
    if (request.method === 'OPTIONS') {
      return new Response(null, { headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET',
        'Access-Control-Allow-Headers': 'Content-Type',
      }});
    }
    const url = new URL(request.url);
    const days = parseInt(url.searchParams.get('days') || '7');
    const now = new Date();
    const start = new Date(now);
    start.setDate(start.getDate() - (days - 1));
    const fmt = d => d.toISOString().split('T')[0];
    const query = `{
      viewer {
        zones(filter: {zoneTag: "${env.CF_ZONE_ID}"}) {
          daily: httpRequests1dGroups(
            limit: ${days} orderBy: [date_ASC]
            filter: {date_geq: "${fmt(start)}", date_leq: "${fmt(now)}"}
          ) {
            dimensions { date }
            sum { pageViews requests countryMap { clientCountryName requests } }
            uniq { uniques }
          }
        }
      }
    }`;
    const res = await fetch('https://api.cloudflare.com/client/v4/graphql', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${env.CF_API_TOKEN}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ query }),
    });
    return new Response(await res.text(), {
      status: res.status,
      headers: {
        'Content-Type': 'application/json',
        'Access-Control-Allow-Origin': '*',
      },
    });
  },
};
